VNC Configuration

~]# yum install tigervnc-server ~]# cp /usr/lib/systemd/system/vncserver@.service /etc/systemd/system/vncserver@.service Edit and add in : /etc/systemd/system/vncserver\@.service ~]# systemctl daemon-reload ~]# su - USER ~]$ vncpasswd Password: Verify:

~]# systemctl start vncserver@:1.service ~]# systemctl enable vncserver@:1.service

~]# iptables -I INPUT 1 -p tcp --dport 5901 -j ACCEPT
