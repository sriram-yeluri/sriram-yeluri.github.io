DEVICE="eth0"
ONBOOT=yes
NETBOOT=yes

HWADDR="00:08:a2:0a:ba:b8"

HWADDR=00:08:A2:0A:BA:B8

DNS2=8.8.8.8

# systemctl restart network

Verify new IP settings: ip a s eth0
