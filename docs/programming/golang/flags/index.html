Golang flags for creating CLI

Defining flags and using them to create a CLI in golang is achieved by following the below steps.

Import flags package

import (
    "flag"
)

Define flags

var name string
var prompt bool

Parse flags

flag.variableType(variable_reference, “flag_name”, “default_value”, “Usage of flag”)

flag.StringVar(&name,"name","","required a name to display")
flag.BoolVar(&prompt,"prompt","","Use prompt to input a name")

/* Call this after all flags are defined */
flag.Parse()

Show usage if flags are invalid

if prompt == false && name == "" {
    flag.Usage()
    os.Exit(1)
}
/* flag.PrintDefaults() can be used to print the usage message explicitly */

Example Code

package main

import (
	"flag"
	"fmt"	
	"os"
)

func main(){
	var name string
	flag.StringVar(&name,"name","","required name flag")
	flag.Parse()

	if name == ""{
		flag.Usage()		
		os.Exit(1)
	} else {
		fmt.Println("Name : ",name)
	}
}
